48template <typename Op> py::object bitwise_core(py::array z2r_1, py::array z2r_2, Op op) {
49 auto buf1 = z2r_1.request();
50 auto buf2 = z2r_2.request();
51
52 if (buf1.itemsize != buf2.itemsize) {
53 throw std::runtime_error("Input arrays must have the same itemsize (dtype compatibility).");
54 }
55 if (buf1.size != buf2.size) {
56 throw std::runtime_error("Input arrays must have the same size.");
57 }
58
59 size_t total_bytes = buf1.size * buf1.itemsize;
60 py::array z2r_out = py::array(z2r_1.dtype(), buf1.shape);
61 auto buf_out = z2r_out.request();
62
63 // Cut the data into 64-bit chunks for faster proceCest ssing
64 // ptr1_64 and ptr2_64 are the pointers to the input data, ptr_out_64 is the pointer to the
65 // output data
66 // TODO: With C++20, we could use std::assume_aligned, which could potentially lead to better
67 // auto-vectorization by the compiler? ? Since NumPy cant guarantee alignement and we cant
68 // afford to copy the data just to get better SIMD, I dont know if assume_aligned is worth
69 // anything.
70 const uint64_t *ptr1_64 = std::bit_cast<uint64_t *>(buf1.ptr);
71 const uint64_t *ptr2_64 = std::bit_cast<uint64_t *>(buf2.ptr);
72 int64_t *ptr_out_64 = std::bit_cast<int64_t *>(buf_out.ptr);
73
74 size_t num_u64_chunks = total_bytes / 8;
75
76#ifdef USE_OPENMP
77 #pragma omp parallel for if (num_u64_chunks >= BOPS_THRESHOLD_PARALLEL) schedule(static)
78#endif
79 for (size_t i = 0; i < num_u64_chunks; ++i) {
80 // Applies the bitwise operation.
81 ptr_out_64[i] = op(ptr1_64[i], ptr2_64[i]);
82 }
83
84 // Handle any bytes that don't fit into a 64-bit chunk (the tail)
85 size_t tail_bytes = total_bytes % 8;
86 if (tail_bytes > 0) {
87 const uint8_t *ptr1_8 = std::bit_cast<uint8_t *>(buf1.ptr);
88 const uint8_t *ptr2_8 = std::bit_cast<uint8_t *>(buf2.ptr);
89 int8_t *ptr_out_8 = std::bit_cast<int8_t *>(buf_out.ptr);
90
91 size_t start_byte = num_u64_chunks * 8;
92 for (size_t i = 0; i < tail_bytes; ++i) {
93 ptr_out_8[start_byte + i] = op(ptr1_8[start_byte + i], ptr2_8[start_byte + i]);
94 }
95 }
96
97 // If the input arrays were actually scalars (shape == ()), return a scalar as well
98 if (buf1.size == 1) {
99 std::vector<ssize_t> shape0{}; // zero-dim
100 std::vector<ssize_t> strides0{}; // must match ndim (0)
101 py::array scalar(z2r_out.dtype(), shape0, strides0, buf_out.ptr,
102 z2r_out); // TODO: Check if this is zero-copy
